    <script>
        // 1. CONFIGURATION
        // Replace these with your actual image URLs
        const imageUrls = [
            "https://images.unsplash.com/photo-1542038784456-1ea8e935640e?w=600&q=80",
            "https://images.unsplash.com/photo-1552168324-d612d77725e3?w=500&q=80",
            "https://images.unsplash.com/photo-1516483638261-f4dbaf036963?w=800&q=80",
            "https://images.unsplash.com/photo-1523906834658-6e24ef2386f9?w=400&q=80",
            "https://images.unsplash.com/photo-1492691527719-9d1e07e534b4?w=600&q=80",
            "https://images.unsplash.com/photo-1500917293891-ef795e70e1f6?w=700&q=80",
            "https://images.unsplash.com/photo-1469334031218-e382a71b716b?w=500&q=80"
        ];

        const container = document.getElementById('scrollContainer');
        const wrapper = document.getElementById('contentWrapper');

        // 2. HELPER TO GENERATE RANDOM LAYOUT
        function createInfoBlock(src, index) {
            const div = document.createElement('div');
            div.classList.add('scrapbook-item');
            
            // Random Width (between 250px and 500px)
            const width = Math.floor(Math.random() * (500 - 250 + 1) + 250);
            div.style.width = `${width}px`;

            // Random Alignment
            const alignments = ['align-left', 'align-center', 'align-right'];
            const randomAlign = alignments[Math.floor(Math.random() * alignments.length)];
            div.classList.add(randomAlign);

            // Random Vertical Offset (to break strict lines)
            const offsets = ['offset-neg', 'offset-pos', ''];
            const randomOffset = offsets[Math.floor(Math.random() * offsets.length)];
            if(randomOffset) div.classList.add(randomOffset);

            const img = document.createElement('img');
            img.src = src;
            div.appendChild(img);

            return div;
        }

        // 3. BUILD THE LISTS
        // We create a "Set" of HTML.
        function generateSet() {
            const fragment = document.createDocumentFragment();
            imageUrls.forEach((url, i) => {
                const item = createInfoBlock(url, i);
                fragment.appendChild(item);
            });
            return fragment;
        }

        // 4. INJECT SETS (CLONING STRATEGY)
        // We need 3 sets: [Top Clone] [Original Middle] [Bottom Clone]
        const setA = document.createElement('div'); 
        setA.style.width = '100%'; setA.style.display = 'flex'; setA.style.flexDirection = 'column';
        setA.appendChild(generateSet());
        
        const setB = setA.cloneNode(true); // The middle "real" set
        const setC = setA.cloneNode(true); // The bottom clone

        wrapper.appendChild(setA);
        wrapper.appendChild(setB);
        wrapper.appendChild(setC);

        // 5. INFINITE SCROLL LOGIC
        let setHeight = 0;

        // Wait for images to load to calculate height correctly
        window.addEventListener('load', () => {
            setHeight = setA.offsetHeight;
            
            // Initial Jump: Start user in the Middle (Top of Set B)
            container.scrollTop = setHeight;

            container.addEventListener('scroll', handleScroll);
        });

        function handleScroll() {
            const scrollTop = container.scrollTop;

            // IF we scroll UP past the top of Set B -> Teleport to bottom of Set B
            if (scrollTop <= 10) { // Using 10px buffer
                container.scrollTop = setHeight + scrollTop; 
            }
            // IF we scroll DOWN past the end of Set B -> Teleport to top of Set B
            else if (scrollTop >= (setHeight * 2)) {
                container.scrollTop = scrollTop - setHeight;
            }
        }

    </script>
